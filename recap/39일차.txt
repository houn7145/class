JavaScript와 Java의 조건문, 삼항연산자는 동일함
짧은 조건문은 논리연산자의 특성을 조건문으로 사용하며 다음 논리합 연산자를 사용한 표현식은 뒤에 어떠한 값이 들어가도 
항상 참이므로 우변을 실행하지 않음
true || OOO // 우변을 항상 실행하지 않음
false || XXX //우변을 항상 실행함

JavaScript 반복문
배열, 배열에서의 반복문과 forEach함수, do~while문, while문

배열
var array = [273, 32, 103, 57, 52];
어떤 자료형도 배열 요소가 될 수 있으며 배열 방의 크기는 가변적임

arr.push(8); // arr 끝번방에 추가
arr.unshift(0); // 제일 앞에 추가
var value = arr.pop(); // 마지막 배열방의 값을 reture하고 삭제
slice는 원본 배열을 바꾸지 않고 새로운 배열을 만듦
splice는 원본 배열을 수정

1. 일반 for문
        document.write('<h3>1. 일반 for문 이용</h3>');
        for(var idx=0 ; idx < arr.length ; idx++){
            document.write(idx + '번째 : ' + arr[idx] + '<br>');
        }

2. for-in문(object | array)
        document.write('<h3>2. for~in문 이용</h3>');
        for(var idx in arr){ // idx변수에 data가 아닌 index가 copy됨
            document.write(idx + '번째 : ' + arr[idx] + '<br>');
        }

3. forEach함수 1
        document.write('<h3>3. forEach문 이용</h3>');
        // 배열명 : arr 
        // 데이터 : 273, 32 '홍길동'
        // index : 0, 1, 2
        arr.forEach(function(data, idx){
            document.write(idx + '번째 : ' + data + '<br>');
        });

4. forEach함수 2 - 배열 반복시 index가 필요 없을 때
        document.write('<h3>4. index없이 forEach문 이용</h3>');
        arr.forEach(function(data){
            document.write(data + '<br>');
        });

JavaScript 함수
익명 함수 - 함수는 코드의 집합을 나타내는 자료형
var functionVar = function() { }; // 함수 표현식
var functionVar = () => {}; // 화살표 함수

매개변수 지정 방법
() => { . . . } // 매개변수가 없을 경우
x =>{ . . . } // 매개변수가 한 개인 경우, 소괄호를 생략할 수 있음
(x, y) => { . . . } // 매개변수가 여러 개인 경우, 소괄호를 생략할 수 없음

함수 몸체 지정 방법
x => { return x*x } // single line block
x => x * x         // 함수 몸체가 한줄의 구문이라면 중괄호를 생략
() => { return { a: 1 }; }
() => ({ a: 1 })  // 위 표현과 동일하다. 객체 반환시 소괄호를 사용함

선언적 함수
※ function 함수명 ( ) { // 함수 선언식}

매개변수
자바 스크립트는 함수를 생성할 때 지정한 매개변수보다 많거나 적은 매개변수를 사용하는 것을 허용하며
원래 함수에서 선언된 매개변수보다 많게 사용하면 일반적으로 추가된 매개변수는 무시함
원래 함수에서 선언한 매개변수보다 적게 사용하면 지정하지 않는 매개변수는 undefined로 입력됨
